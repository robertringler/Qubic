Quick Start Example

Python 
from quasim import QuantumSimulator
from quasim.circuits import QuantumCircuit

# Initialize 32-qubit simulation on 4 GPUs
sim = QuantumSimulator(
    qubits=32,
    backend='cuQuantum',
    gpus=[0, 1, 2, 3],
    precision='fp32'
)

# Build quantum circuit
circuit = QuantumCircuit(32)
circuit.h(0)  # Hadamard gate on qubit 0
for i in range(31):
    circuit.cx(i, i+1)  # CNOT cascade

# Execute and measure
result = sim.run(circuit, shots=1024)
print(f"Fidelity: {result.fidelity:.4f}")
print(f"Execution time: {result.time_ms:.2f}ms")
